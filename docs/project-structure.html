your-project/
├── api.yaml                 # Endpoint tanımları (kaynak)
├── api/
│   └── index.go             # Vercel serverless giriş noktası
├── cmd/
│   ├── gofire/main.go       # CLI aracı
│   └── server/main.go       # Yerel sunucu giriş noktası
├── config/
│   ├── env.go               # Ortam yapılandırması
│   └── version.go           # Versiyon sabiti
├── handlers/
│   ├── health.go            # Dahili health handler
│   ├── root.go              # Dahili root sayfa
│   └── *.go                 # Üretilen handler stub'ları
├── cache/
│   ├── cache.go             # Cache arayüzü
│   └── upstash.go           # Upstash Redis implementasyonu
├── middleware/
│   └── firebase.go          # Firebase Auth middleware
├── server/
│   └── server.go            # Chi router (gofire gen ile üretilir)
├── internal/
│   ├── scaffold/            # Kod üretim motoru
│   └── yaml/                # api.yaml parser
├── vercel.json
├── .env.example
└── go.mod
